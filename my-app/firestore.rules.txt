// Firestore Security Rules для React Store
// Скопируйте это в Firebase Console: Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Функция для проверки, является ли пользователь администратором
    function isAdmin() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isAdmin == true;
    }

    // Функция для проверки, авторизован ли пользователь
    function isAuthenticated() {
      return request.auth != null;
    }

    // Функция для проверки, является ли пользователь владельцем документа
    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    // Коллекция users - профили пользователей
    match /users/{userId} {
      // Пользователь может читать свой профиль
      allow read: if isOwner(userId);
      
      // Администратор может читать все профили
      allow read: if isAdmin();
      
      // Пользователь может писать в свой профиль
      allow write: if isOwner(userId);
      
      // Администратор может писать все профили
      allow write: if isAdmin();
      
      // Создание нового пользователя при регистрации
      allow create: if request.auth.uid == userId;
    }

    // Коллекция products - товары
    match /products/{document=**} {
      // Все могут читать товары
      allow read: if true;
      
      // Только администраторы могут писать товары
      allow write: if isAuthenticated() && isAdmin();
    }

    // Коллекция orders - заказы
    match /orders/{orderId} {
      // Пользователь может читать свои заказы
      allow read: if isOwner(resource.data.userId);
      
      // Администратор может читать все заказы
      allow read: if isAdmin();
      
      // Пользователь может создавать свои заказы
      allow create: if isAuthenticated() && request.resource.data.userId == request.auth.uid;
      
      // Администратор может обновлять статус заказа
      allow update: if isAdmin();
    }

    // Коллекция cart - корзины пользователей
    match /cart/{userId}/{document=**} {
      // Пользователь может читать и писать свою корзину
      allow read, write: if isOwner(userId);
    }

    // Коллекция settings - глобальные настройки (только для администраторов)
    match /settings/{document=**} {
      allow read: if true;
      allow write: if isAdmin();
    }
  }
}
